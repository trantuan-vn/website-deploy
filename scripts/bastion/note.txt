# Nguyên nhân: Bastion đặt trong subnet API (/28) nhưng security list thiếu egress TCP/6443
#   tới API subnet. SSH tunnel mở port local OK nhưng không forward được tới API.
# Giải pháp 1 (khuyến nghị): setup-bastion.sh đặt bastion trong oke-nodesubnet
# Giải pháp 2: connect-oke.sh tự thêm egress rule nếu bastion vẫn ở API subnet
# Giải pháp 3: Xóa bastion cũ, chạy lại ./setup-bastion.sh

kill <PID> hoặc pkill -f "ssh.*bastion"

# Xóa session
oci bastion session delete --session-id $SESSION_ID --force
